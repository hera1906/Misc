    -- Xmonad
        M-C-r           -- Recompiles xmonad
        M-S-r           -- Restarts xmonad
        M-S-q           -- Quits xmonad

    -- Terminal
        M-<Return>      -- Terminal

    --- Dmenu Scripts (Alt+Ctr+Key)
        M-S-<Return>    -- Dmenu run

    -- Prompts
        M-S-<Return>    -- Shell Prompt
        M-S-o           -- Xmonad Prompt
        M-S-s           -- Ssh Prompt
        M-S-m           -- Manpage Prompt

    -- Windows
        M-S-c           -- Kill the currently focused client
        M-S-a           -- Kill all windows on current workspace

    -- Floating windows
        M-f             -- Toggles floats layout
        M-<Delete>      -- Push floating window back to tile
        M-S-<Delete>    -- Push ALL floating windows to tile

    -- Grid Select
        M-S-t           -- Grid spawn selected'
        M-S-g           -- Grid goto selected
        M-S-b           -- Gridbring selected

    -- Windows navigation
        M-m             -- Move focus to the master window
        M-j             -- Move focus to the next window
        M-k             -- Move focus to the prev window
        M-S-j           -- Swap focused window with next window
        M-S-k           -- Swap focused window with prev window
        M-<Backspace>   -- Moves focused window to master, others maintain order
        M-C-s

        M-<Up>          --  Move focused window to up
        M-<Down>        --  Move focused window to down
        M-<Right>       --  Move focused window to right
        M-<Left>        --  Move focused window to left
        M-S-<Up>        --  Increase size of focused window up
        M-S-<Down>      --  Increase size of focused window down
        M-S-<Right>     --  Increase size of focused window right
        M-S-<Left>      --  Increase size of focused window left
        M-C-<Up>        --  Decrease size of focused window up
        M-C-<Down>      --  Decrease size of focused window down
        M-C-<Right>     --  Decrease size of focused window right
        M-C-<Left>      --  Decrease size of focused window left

    -- Layouts
        M-<Tab>         -- Switch to next layout
        M-<Space>       -- Toggles noborder/full
        M-S-<Space>     -- Toggles struts
        M-S-n           -- Toggles noborder
        M-S-x           -- Togglex Reflect X
        M-S-y           -- Togglex Reflect Y

        M-h             -- Shrink
        M-l             -- Expand
        M-C-j           -- Mirror Shrink
        M-C-k           -- Mirror Expand
        M-S-;           -- Zoom Reset
        M-;             -- Zoom full toggle

    -- Workspaces
        M-.             -- Switch focus to next monitor
        M-,             -- Switch focus to prev monitor

    -- Scratchpads
        M-C-<Return>    -- Scratchpad Terminal
        M-C-c           -- Scratchpad cmus
